获取当前最顶层视图

方法一

+ (UIViewController *)getCurrentVC{
    UIViewController *result = nil;
    UIWindow * window = [[UIApplication sharedApplication] keyWindow];
    if (window.windowLevel != UIWindowLevelNormal)
    {
        NSArray *windows = [[UIApplication sharedApplication] windows];
        for(UIWindow * tmpWin in windows)
        {
            if (tmpWin.windowLevel == UIWindowLevelNormal)
            {
                window = tmpWin;
                break;
            }
        }
    }
    id  nextResponder = nil;
    UIViewController *appRootVC=window.rootViewController;
    if (appRootVC.presentedViewController) {
        nextResponder = appRootVC.presentedViewController;
    }else{
        NSArray *views = [window subviews];
        if (views.count == 0) {
            return nil;
        }
        UIView *frontView = [[window subviews] objectAtIndex:0];
        nextResponder = [frontView nextResponder];
    }
    if ([nextResponder isKindOfClass:[UITabBarController class]]){
        UITabBarController * tabbar = (UITabBarController *)nextResponder;
        UINavigationController * nav = (UINavigationController *)tabbar.viewControllers[tabbar.selectedIndex];
        result=nav.childViewControllers.lastObject;
    }else if ([nextResponder isKindOfClass:[UINavigationController class]]){
        UIViewController * nav = (UIViewController *)nextResponder;
        result = nav.childViewControllers.lastObject;
    }else{
        result = nextResponder;
    }
    return result;
}

方法二

- (UINavigationController *)navigationViewController
{
    if ([self.window.rootViewController isKindOfClass:[BaseNavViewController class]])
    {
        return (BaseNavViewController *)self.window.rootViewController;
    }else if ([self.window.rootViewController isKindOfClass:[UITabBarController class]])
    {
        UIViewController *selectVc = [((UITabBarController *)self.window.rootViewController) selectedViewController];
        if ([selectVc isKindOfClass:[UINavigationController class]])
        {
            return (UINavigationController *)selectVc;
        }
    }
    else
    {
        UIWindow *window = [[[UIApplication sharedApplication] delegate]window];
        HomeTabBarViewController *nav = (HomeTabBarViewController *)window.rootViewController;
        if ([[nav class] isEqual:[HomeTabBarViewController class]]) {
            BaseNavViewController *navi = nav.childViewControllers[0];
            if ([[navi class]isEqual:[BaseNavViewController class]]) {
                return navi;
            }
        }
    }
    return nil;
}

- (UIViewController *)topViewController
{
    UINavigationController *nav = [self navigationViewController];
    return nav.topViewController;
}

/**
 *  延迟执行
 *
 *  @param block 执行的block
 *  @param delay 延迟的时间：秒
 */
- (void)performBlock:(void(^)(void))block afterDelay:(NSTimeInterval)delay {
    dispatch_after(dispatch_time(DISPATCH_TIME_NOW, (int64_t)(delay * NSEC_PER_SEC)), dispatch_get_main_queue(), ^{
        block();
    });
}

UIScrollView、UITableView、UICollectionView 下拉与顶部空白颜色填充问题

- (void)buildColorView {

  // 创建 tableView 之后，新建一个 UIView 加到 tableView 上
    _colorView = [[UIView alloc] initWithFrame:CGRectMake(0, -ScreenHeight, kScreenWidth, ScreenHeight)];  // 高度随便，只要下拉的高度不超出View的高度就行了
    _colorView.backgroundColor = kMainColor;
    [self.tableView addSubview:_colorView];
}

// 代理
- (void)scrollViewDidScroll:(UIScrollView *)scrollView {
    CGFloat y = scrollView.contentOffset.y;
    if (scrollView == _tableView) {
        if (y < -ScreenHeight) {
            CGRect frame = _colorView.frame;
            frame.size.height = -y;
            frame.origin.y = y;
            _colorView.frame = frame;
        }
    }
}

推送在杀掉应用的情况下，如何实现页面跳转

  1. 在didFinishLaunchingWithOptions方法中存取推送信息

    NSDictionary *dictionary = [launchOptions objectForKey:UIApplicationLaunchOptionsRemoteNotificationKey];
if (dictionary) {
    NSDictionary *userInfo = @{@"messagetype":[NSString stringIsExist:dictionary[@"messagetype"]],
                               @"messageid":[NSString stringIsExist:dictionary[@"id"] ],
                               @"newstype":[NSString stringIsExist:dictionary[@"type"]],
                               @"pushid":[NSString stringIsExist:dictionary[@"pushid"]],
                               @"is_login":[NSString stringIsExist:dictionary[@"is_login"]],
                               @"is_share":[NSString stringIsExist:dictionary[@"is_share"]],
                               @"linkId":[NSString stringIsExist:dictionary[@"linkId"] ]
    };
    [CommonUtils saveObjectToUD:userInfo key:@"PUSHNOTICE"];
}

  2. 推送代理方法中，判断是正常推送还是杀掉应用的情况

    if ([CommonUtils getInfoObject:@"PUSHNOTICE"]) {
    return;
}

  3. 在应用主页添加跳转

    - (void)viewDidAppear:(BOOL)animated{
    [super viewDidAppear:animated];
    if ([CommonUtils getInfoObject:@"PUSHNOTICE"] && [[CommonUtils getInfoObject:@"PUSHNOTICE"] isKindOfClass:[NSDictionary class]]) {
        NSDictionary *dict = [CommonUtils getInfoObject:@"PUSHNOTICE"];
        [self selectMessagetypeView:dict[@"messagetype"] andid:dict[@"messageid"] newstype:dict[@"newstype"] pushid:dict[@"pushid"] is_login:dict[@"is_login"] is_share:dict[@"is_share"] linkId:dict[@"linkId"]];
    }
}

UIButton倒计时按钮（闪烁问题也轻松解决）

很多应用中发短信倒计时功能，一般都会将NSTimer与UIButton结合来实现此功能，如果UIButton是这么初使化的：在测试时会发现，当定时器每隔一秒更改标题时，会有闪烁现象，将UIButtonType.System更改为UIButtonType.Custom即可
